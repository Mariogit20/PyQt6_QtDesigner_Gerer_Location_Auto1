# Méthode 2: Chargement dynamique à l'exécution

# Cette méthode charge le fichier .ui directement à l'exécution sans le convertir au préalable. Elle est utile pour des projets où l'interface est susceptible de changer fréquemment, mais elle est légèrement moins performante que la première méthode.

#     Importez la bibliothèque uic de PyQt.

#     Chargez le fichier .ui en utilisant la méthode loadUi.

import sys
from PyQt6.QtWidgets import QApplication, QMainWindow
from PyQt6.uic import loadUi

class MonApplication(QMainWindow):
    def __init__(self):
        super().__init__()
        # Chargez le fichier .ui
        loadUi("votre_fichier.ui", self)

        # Vous pouvez maintenant accéder aux widgets par leur nom d'objet
        # self.bouton_enregistrer.clicked.connect(...)
        # self.champ_texte.setText(...)

if __name__ == "__main__":
    app = QApplication(sys.argv)
    fenetre = MonApplication()
    fenetre.show()
    sys.exit(app.exec())


# Note : Pour que cette méthode fonctionne, le fichier .ui doit être accessible depuis l'emplacement où votre script Python est exécuté.
